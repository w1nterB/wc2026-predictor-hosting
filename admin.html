Database setup SQL
Run this once in Supabase → SQL Editor to create all required tables:
-- Run in Supabase SQL Editor → New query
-- ⚠️ Also run the users table SQL shown in Admin → Database connection

-- USERS TABLE (for self-service auth):
create table if not exists users (
  id         bigserial primary key,
  email      text unique not null,
  name       text not null,
  pin_hash   text not null,
  fn         text default 'di',
  created_at timestamptz default now()
);
alter table users enable row level security;
create policy "anon all users" on users for all using (true) with check (true);
create index if not exists users_email_idx on users(email);
create index if not exists users_name_idx  on users(name);

-- The other tables:
-- ALSO CREATE (paste in same SQL editor run):
-- See match_previews and did_you_know tables below

create table if not exists match_previews (
  match_id      text primary key,
  first_meeting boolean default false,
  h2h           jsonb default '[]',
  fact          text default '',
  stat          text default '',
  venue         text default ''
);
alter table match_previews enable row level security;
create policy "anon all match_previews" on match_previews for all using (true) with check (true);

create table if not exists did_you_know (
  id serial primary key, fact text, category text
);
alter table did_you_know enable row level security;
create policy "anon all dyk" on did_you_know for all using (true) with check (true);

-- The other tables:

create table if not exists config (
  key   text primary key,
  value text
);

create table if not exists participants (
  name text primary key,
  fn   text,
  pin  text
);

create table if not exists results (
  match_id   text primary key,
  home_score integer,
  away_score integer
);

create table if not exists bracket (
  match_id   text primary key,
  home_name  text,
  away_name  text,
  home_score integer,
  away_score integer
);

create table if not exists predictions (
  name        text,
  type        text,
  match_id    text default '',
  home_score  integer,
  away_score  integer,
  winner      text,
  runner_up   text,
  top_scorer  text,
  surprise    text,
  sf1 text, sf2 text, sf3 text, sf4 text,
  submitted_at timestamptz default now(),
  primary key (name, type, match_id)
);

-- Row Level Security — allow public read/write via anon key
alter table config       enable row level security;
alter table participants enable row level security;
alter table results      enable row level security;
alter table bracket      enable row level security;
alter table predictions  enable row level security;

create policy "anon all" on config       for all using (true) with check (true);
create policy "anon all" on participants for all using (true) with check (true);
create policy "anon all" on results      for all using (true) with check (true);
create policy "anon all" on bracket      for all using (true) with check (true);
create policy "anon all" on predictions  for all using (true) with check (true);
